private void agradecimientos() {
			JLabel[] agradecimientos = new JLabel[15];
			for (int i = 0; i < 15; i++) {
				agradecimientos[i] = new JLabel();
				agradecimientos[i].setVisible(true);
				agradecimientos[i].setForeground(new Color(255,0,0));
				agradecimientos[i].setFont(new Font("Sitka Text", Font.BOLD, 18));
				agradecimientos[i].setBackground(new Color(255,255,255));
				agradecimientos[i].setBounds((int) (gui.getFrameWidth() * 0.2), (int) (gui.getFrameHeight() * 0.026) + i * 17, (int) (gui.getFrameWidth() * 0.5), (int) (gui.getFrameHeight() * 0.4));
			}
			
			agradecimientos[0].setText("¡AGRADECIMIENTOS!");
			agradecimientos[1].setText("Las siguientes personas nos ayudaron notablemente en el proceso de desarrollo de este juego:");
			
			try {
				BufferedReader br;
				FileReader fr = new FileReader("./bin/Agradecimientos.txt");
				br = new BufferedReader(fr);
				
				for (int i = 2; i < 15; i++)
					agradecimientos[i].setText("+ " + br.readLine());
				
				br.close();
			}
			catch(IOException exc) {
				System.out.println("Error al leer el archivo de Agradecimientos.");
				exc.printStackTrace();
			}
			
			for(int i = 0; i < 15; i++)
				gui.getPanel().add(agradecimientos[i]);
		}
